Create PROCEDURE [dbo].[tblLoaiTin_Insert]
@Ten NVarChar(256),
@Cha Int,
@ThuTu Int,
@NgayTao DateTime,
@IconLoaiTin VarChar(50),
@NguonTin Int,
@Lang NChar(5),
@TinhTrang Bit
as
insert into tblLoaiTin
([Ten],[Cha],[ThuTu],[NgayTao],[IconLoaiTin],[NguonTin],[Lang],[TinhTrang])
Values
(@Ten,@Cha,@ThuTu,@NgayTao,@IconLoaiTin,@NguonTin,@Lang,@TinhTrang)
go
Create PROCEDURE [dbo].[tblLoaiTin_Update]
@ID Int,
@Ten NVarChar(256),
@Cha Int,
@ThuTu Int,
@NgayTao DateTime,
@IconLoaiTin VarChar(50),
@NguonTin Int,
@Lang NChar(5),
@TinhTrang Bit
as
Update tblLoaiTin Set
 [Ten]= @Ten,[Cha]= @Cha,[ThuTu]= @ThuTu,[NgayTao]= @NgayTao,[IconLoaiTin]= @IconLoaiTin,[NguonTin]= @NguonTin,[Lang]= @Lang,[TinhTrang]= @TinhTrang
 Where ([ID]= @ID)
go
Create PROCEDURE [dbo].[tblLoaiTin_Delete]
@ID Int
as
Delete tblLoaiTin
 Where  ([ID]= @ID)
go
Create PROCEDURE [dbo].[tblLoaiTin_SelectAll]
as
Select * from tblLoaiTin
go
Create PROCEDURE [dbo].[tblLoaiTin_SelectByID]
@ID Int
as
Select * from tblLoaiTin
 Where  ([ID]= @ID)
go
